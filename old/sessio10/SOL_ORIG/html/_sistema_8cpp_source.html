31 void Sistema::luchas_org_cola(queue<Organismo> &c, Organismo &o, bool &sobrevive)
32 {
33 // Busca el primer organismo de la cola c que consigue matar al organismo o y
34 // borra (y cuenta) todos los que resultan destruidos por éste. Para que el
35 // borrado sea efectivo, los supervivientes se van añadiendo al final de la
36 // cola. Al final, los que no hayan luchado se pasan al final la cola mediante
37 // una nueva operación privada auxiliar: "recolocar".
38 
39  Organismo actual;
40  int resultado;
41 
42  sobrevive = true;
43  int long_cola = c.size();
44 
45 // Inv: sobrevive = ninguno de los elementos visitados de C destruye a o;
46 // c = elementos no visitados de C seguidos por los elementos visitados
47 // con su contador de víctimas actualizado, excepto los que hayan muerto
48 // al enfrentarse con o, en el mismo orden en que estaban en C;
49 // long_cola = número de elementos no visitados de C;
50 // o tiene actualizado su contador de víctimas
51 //
52 // Cota: long_cola
53 
54  while (long_cola>0 and sobrevive) {
55  actual = c.front();
56  resultado = o.lucha_organismos(actual);
57  switch (resultado) {
58  case 0: { // los dos mueren
59  sobrevive = false; // no actualizamos las víctimas de actual
60  o.incrementar_victimas(); // porque éste ya no pertenece al sistema
61  break;
62  }
63  case 1: { // o muere, actual sobrevive
64  sobrevive= false;
65  actual.incrementar_victimas();
66  c.push(actual);
67  break;
68  }
69  case 2: { // o sobrevive, actual muere
70  o.incrementar_victimas();
71  break;
72  }
73  case 3: { // los dos sobreviven
74  c.push(actual);
75  break;
76  }
77  }
78  c.pop();
79  --long_cola;
80  }
81 
82 // Post1: Inv1 y (long_cola == 0 or not sobrevive)
83 
84 // Falta pasar al final de c los elementos no visitados de C (por Inv1, son
85 // los long_cola primeros de c), para mantener el orden por identificador
86 
87  recolocar(long_cola,c);
88 }

109 void Sistema::leer(int N)
110 {
111 // Esta operación actualiza todos los componentes de un sistema, leyéndolos
112 // del canal estándar; ofrece la posiblidad de leer y almacenar algunos
113 // organismos defensivos. El número de éstos se proporciona al comienzo de
114 // la operación.
115 
116  clear(def);
117  clear(mal);
118  int num = readint();
119  Organismo o;
120  for (int i = 1; i <= num; ++i) {
121  o.leer(N);
122  o.anadir_id(i);
123  if (o.es_maligno()) mal.push(o);
124  else def.push(o);
125  }
126  id = num+1;
127 }
