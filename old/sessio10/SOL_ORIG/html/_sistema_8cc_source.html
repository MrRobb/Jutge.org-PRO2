00031 void Sistema::luchas_org_cola(queue<Organismo> &c, Organismo &o, bool &sobrevive)
00032 {
00033 // Busca el primer organismo de la cola c que consigue matar al organismo o y
00034 // borra (y cuenta) todos los que resultan destruidos por éste. Para que el
00035 // borrado sea efectivo, los supervivientes se van añadiendo al final de la 
00036 // cola. Al final, los que no hayan luchado se pasan al final la cola mediante
00037 // una nueva operación privada auxiliar: "recolocar".
00038 
00039    Organismo actual; 
00040    int resultado; 
00041  
00042    sobrevive = true;
00043    int long_cola = c.size();
00044   
00045 // Inv: sobrevive = ninguno de los elementos visitados de C destruye a o;
00046 //      c = elementos no visitados de C seguidos por los elementos visitados
00047 //      con su contador de víctimas actualizado, excepto los que hayan muerto 
00048 //      al enfrentarse con o, en el mismo orden en que estaban en C;
00049 //      long_cola = número de elementos no visitados de C;
00050 //      o tiene actualizado su contador de víctimas
00051 // 
00052 // Cota: long_cola
00053 
00054    while (long_cola>0 and sobrevive) {
00055       actual = c.front();
00056       resultado = o.lucha_organismos(actual);
00057       switch (resultado) {
00058          case 0: {                    // los dos mueren
00059             sobrevive = false;        // no actualizamos las víctimas de actual 
00060             o.incrementar_victimas(); // porque éste ya no pertenece al sistema
00061             break;
00062          } 
00063          case 1: {                    // o muere, actual sobrevive
00064             sobrevive= false;
00065             actual.incrementar_victimas();

00067             break;
00068          }
00069          case 2: {                    // o sobrevive, actual muere
00070             o.incrementar_victimas();
00071             break;
00072          }
00073          case 3: {                    // los dos sobreviven
00074             c.push(actual);
00075             break;
00076          }
00077       }
