00030 int Organismo::lucha_organismos(const Organismo &o2) const 
00031 {
00032   // Ésta es la operación más importante del módulo. Dados dos organismos, hay
00033   // que decidir primero si van a luchar de verdad o no, es decir, si sus
00034   // estructuras celulares son simétricas o no. Por eso, introducimos la
00035   // operación auxiliar "simetricos" en la parte privada.  Notad que en la
00036   // cabecera de la operación "simetricos" decimos que los parámetros han de
00037   // ser árboles de Celula, pero la operación es independiente del tipo de los
00038   // elementos del árbol, ya que éstos nunca se llegan a consultar.  Asímismo,
00039   // debemos disponer de otra operación que aplique las luchas de las células
00040   // de dos árboles, sabiendo que son simétricos. Aquí sí es relevante el
00041   // hecho de que los árboles sean de células. Esta nueva operación privada se
00042   // llama "lucha_arboles".
00043 
00044   int n;
00045   Arbre<Celula> aux1(celulas);       // podemos prescindir de aux1 y aux2
00046   Arbre<Celula> aux2(o2.celulas);    // si simetricos y lucha_arboles
00047   if (simetricos(aux1, aux2)) {      // replantan sus parámetros para no
00048     aux1 = celulas;                  // destruirlos, pero entonces 
00049     aux2 = o2.celulas;               // lucha_organismos no puede ser const
00050     pair<int, int> m = lucha_arboles(aux1, aux2);
00051     if (m.first == m.second) n = 0;
00052     else if (m.first < m.second) n = 1;
00053     else n = 2; // m.first > m.second
00054   }
00055   else n = 3;
00056   return n;
00057 }
